# Q6 Answer template

def solution(arr):
    if len(arr) == 1:
        arr[0] = -1
    else:
        arr.sort(reverse = True)
        mi = arr.pop()
        if min(arr) == mi: # 중복처리 부분
            solution (arr)
        else:
            return arr
    return arr

arr = [4, 3, 2, 1,1,1]
answer = solution(arr)
print(answer)
